Loop Statements

While Loop

Execute statements while a condition remains true.

while (condition) { /* statements */ }

For Loop

Traditional C-style for loops with initialization, condition, and increment.

for (init; condition; increment) { /* statements */ }

Do-While Loop

Execute statements at least once, then continue while condition is true.

do { /* statements */ } while (condition);

Loop Examples

loops.csc 
println("=== Loop Examples ===");

// While loop

    print("T-minus ");
    println(countdown);
    countdown = countdown - 1;
}
println("Liftoff!");

// For loop
println("For loop (counting up):");
for (let i: i32 = 1; i <= 3; i = i + 1) {
    print("Count: ");
    println(i);
}

// Do-while loop
println("Do-while loop:");
let attempts: i32 = 0;
do {
    attempts = attempts + 1;
    print("Attempt: ");
    println(attempts);
} while (attempts < 2);

Factorial Calculator

factorial.csc 
let n: i32 = 5;
let factorial: i32 = 1;
let counter: i32 = 1;

print("Calculating factorial of ");
println(n);

while (counter <= n) {
    factorial = factorial * counter;
    print("Step ");
    print(counter);
    print(": ");
    println(factorial);
    counter = counter + 1;
}

print("Final result: ");
print(n);
print("! = ");
println(factorial);

Prime Number Checker

prime.csc 
let testNum: i32 = 17;
let divisor: i32 = 2;
let isPrime: i32 = 1; // 1 = true, 0 = false

print("Testing if ");
print(testNum);
println(" is prime...");

if (testNum <= 1) {
    isPrime = 0;
} else {
    while (divisor * divisor <= testNum) {
        let remainder: i32 = testNum % divisor;
        if (remainder == 0) {
            isPrime = 0;
            print("Found divisor: ");
            println(divisor);
        }
        divisor = divisor + 1;
    }
}

if (isPrime == 1) {
    print(testNum);
    println(" is PRIME!");
} else {
    print(testNum);
    println(" is NOT prime");
}
