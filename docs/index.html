A TypeScript-inspired language that compiles to efficient native code

Static typing with compile-time type checking

Familiar Syntax

TypeScript-inspired syntax for easy adoption

Getting Started

Installation

1

Clone Repository

git clone https://github.com/your-org/cypescript.git
cd cypescript
2

Setup Dependencies (macOS)

./setup-macos.sh
3

Build Compiler

./build.sh

Your First Program

Create a file named hello.csc and try this example:

hello.csc

Control Flow

If Statements

Execute code conditionally based on boolean expressions.

If-Else Examples

control-flow.csc 
let score: i32 = 85;

print("Score: ");
println(score);

// Simple if-else
if (score >= 90) {
    println("Grade: A - Excellent!");
} else {
    if (score >= 80) {
        println("Grade: B - Good work!");
    } else {
        if (score >= 70) {
            println("Grade: C - Satisfactory");
        } else {
            println("Grade: D or F - Needs improvement");
        }
    }
}

// Complex conditions
let bonus: i32 = 5;
let finalScore: i32 = score + bonus;

if (finalScore > score) {
    print("Bonus applied! Final score: ");
    println(finalScore);
}

Loop Statements

While Loop

Execute statements while a condition remains true.

while (condition) { /* statements */ }

For Loop

Traditional C-style for loops with initialization, condition, and increment.

for (init; condition; increment) { /* statements */ }

Do-While Loop

Execute statements at least once, then continue while condition is true.

do { /* statements */ } while (condition);

Loop Examples

loops.csc 
println("=== Loop Examples ===");

// While loop
println("While loop (countdown):");
let countdown: i32 = 3;
while (countdown > 0) {
    print("T-minus ");
    println(countdown);
    countdown = countdown - 1;
}
println("Liftoff!");

// For loop
println("For loop (counting up):");
for (let i: i32 = 1; i <= 3; i = i + 1) {
    print("Count: ");
    println(i);
}

// Do-while loop
println("Do-while loop:");
let attempts: i32 = 0;
do {
    attempts = attempts + 1;
    print("Attempt: ");
    println(attempts);
} while (attempts < 2);

Complete Examples

Factorial Calculator

factorial.csc 
let n: i32 = 5;
let factorial: i32 = 1;
let counter: i32 = 1;

print("Calculating factorial of ");
println(n);

while (counter <= n) {
    factorial = factorial * counter;
    print("Step ");
    print(counter);
    print(": ");
    println(factorial);
    counter = counter + 1;
}

print("Final result: ");
print(n);
print("! = ");
println(factorial);

Prime Number Checker

prime.csc 
let testNum: i32 = 17;
let divisor: i32 = 2;
let isPrime: i32 = 1; // 1 = true, 0 = false

print("Testing if ");
print(testNum);
println(" is prime...");

if (testNum <= 1) {
    isPrime = 0;
} else {
    while (divisor * divisor <= testNum) {
        let remainder: i32 = testNum % divisor;
        if (remainder == 0) {
            isPrime = 0;
            print("Found divisor: ");
            println(divisor);
        }
        divisor = divisor + 1;
    }
}

if (isPrime == 1) {
    print(testNum);
    println(" is PRIME!");
} else {
    print(testNum);
    println(" is NOT prime");
}

Fibonacci Sequence

fibonacci.csc 
let limit: i32 = 8;
let fib1: i32 = 0;
let fib2: i32 = 1;
let count: i32 = 0;

print("Fibonacci sequence (first ");
print(limit);
println(" numbers):");

while (count < limit) {
    if (count == 0) {
        print("F(0) = ");
        println(fib1);
    } else {
        if (count == 1) {
            print("F(1) = ");
            println(fib2);
        } else {
            let nextFib: i32 = fib1 + fib2;
            print("F(");
            print(count);
            print(") = ");
            println(nextFib);
            
            fib1 = fib2;
            fib2 = nextFib;
        }
    }
    count = count + 1;
}
